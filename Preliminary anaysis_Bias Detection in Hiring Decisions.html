import numpy as np
import pandas as pd

# Set random seed for reproducibility
np.random.seed(42)

# Number of applicants
n = 1000

# Generate applicant features
age = np.random.randint(20, 60, size=n)  # Age between 20 and 60
experience = np.random.randint(1, 40, size=n)  # Years of experience
education = np.random.choice(['High School', 'Bachelor', 'Master', 'PhD'], size=n, p=[0.3, 0.4, 0.2, 0.1])
skill_score = np.random.normal(loc=70, scale=10, size=n)  # Normally distributed skill score out of 100

# Encode education levels numerically for the model
education_map = {'High School': 1, 'Bachelor': 2, 'Master': 3, 'PhD': 4}
education_encoded = np.array([education_map[edu] for edu in education])  # Convert to NumPy array

# Create a bias factor penalizing older applicants
bias_factor = -0.03 * age  # Older applicants get slightly lower scores

# Calculate ATS scores (with bias)
ats_score = 0.4 * experience + 0.3 * education_encoded + 0.3 * skill_score + bias_factor

# Assign "Hire" decision based on ATS score threshold
hiring_decision = (ats_score > np.percentile(ats_score, 60)).astype(int)  # Top 40% hired

# Create DataFrame
data = pd.DataFrame({
    'Age': age,
    'Experience': experience,
    'Education': education,
    'SkillScore': skill_score,
    'ATSScore': ats_score,
    'HiringDecision': hiring_decision
})

from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.preprocessing import OneHotEncoder
import tensorflow as tf


df = data.copy()

# Encode education using one-hot encoding
education_encoded = pd.get_dummies(df['Education'], prefix='Education')

# Merge one-hot encoding to the dataset and drop the original education column
df = pd.concat([df, education_encoded], axis=1)
df.drop(['Education'], axis=1, inplace=True)

# Define features and target
features = df[['Age', 'Experience', 'SkillScore'] + list(education_encoded.columns)]
target = df['HiringDecision']

# Split data into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(features, target, test_size=0.2, random_state=42)

# Normalize the numerical features
scaler = StandardScaler()
X_train_scaled = scaler.fit_transform(X_train)
X_test_scaled = scaler.transform(X_test)
